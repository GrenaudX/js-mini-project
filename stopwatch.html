Stopwatch

Track time precisely with a clean and stylish stopwatch.

00:00:00.000

JavaScript Code


let startTime;
let running = false;
let interval;

function startStopwatch() {
  if (!running) {
    startTime = Date.now() - (window.elapsedTime || 0);
    interval = setInterval(updateDisplay, 10);
    running = true;
  }
}

function stopStopwatch() {
  if (running) {
    clearInterval(interval);
    window.elapsedTime = Date.now() - startTime;
    running = false;
  }
}

function resetStopwatch() {
  clearInterval(interval);
  document.getElementById("display").textContent = "00:00:00.000";
  running = false;
  window.elapsedTime = 0;
}

function updateDisplay() {
  const elapsed = Date.now() - startTime;
  let ms = elapsed % 1000;
  let totalSeconds = Math.floor(elapsed / 1000);
  let seconds = totalSeconds % 60;
  let minutes = Math.floor(totalSeconds / 60) % 60;
  let hours = Math.floor(totalSeconds / 3600);

  const format = (num, digits = 2) => num.toString().padStart(digits, "0");

  document.getElementById("display").textContent = 
    `${format(hours)}:${format(minutes)}:${format(seconds)}.${format(ms, 3)}`;
}
